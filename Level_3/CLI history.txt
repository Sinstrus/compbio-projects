#!/bin/bash

# Check if hg38.fa already exists
if [ -f "hg38.fa" ]; then
    echo "hg38.fa already exists. Skipping download."
else
    echo "Downloading hg38.fa.gz (approx 900MB)..."
    # Download from UCSC
    wget https://hgdownload.soe.ucsc.edu/goldenPath/hg38/bigZips/hg38.fa.gz
    
    echo "Unzipping..."
    gunzip hg38.fa.gz
    
    # Create the index file (.fai) immediately to save time later
    # (pyfaidx will do this automatically, but good to ensure it works now)
    echo "Indexing genome (this might take a minute)..."
    python -c "import pyfaidx; pyfaidx.Fasta('hg38.fa')"
    
    echo "Done! Genome ready."
fi
